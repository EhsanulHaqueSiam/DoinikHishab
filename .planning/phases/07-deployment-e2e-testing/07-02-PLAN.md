    **6. Create .github/workflows/e2e.yml (per D-07):**
    Follow the existing build.yml patterns for consistency. Use ubuntu-latest with KVM acceleration. Build debug APK (faster than release, mock data works in debug per D-08). Use reactivecircus/android-emulator-runner@v2 with API level 30, x86_64 arch. Install Maestro via dniHze/maestro-test-action@v1.

    ```yaml
    name: E2E Tests

    on:
      push:
        branches: [main, master]
        paths-ignore:
          - '*.md'
          - '.claude/**'
          - '.planning/**'
      workflow_dispatch:

    env:
      NODE_VERSION: '20'
      BUN_VERSION: 'latest'
      JAVA_VERSION: '17'

    jobs:
      e2e:
        runs-on: ubuntu-latest
        timeout-minutes: 45
        steps:
          - uses: actions/checkout@v4

          - uses: actions/setup-java@v4
            with:
              java-version: ${{ env.JAVA_VERSION }}
              distribution: temurin

          - uses: oven-sh/setup-bun@v2
            with:
              bun-version: ${{ env.BUN_VERSION }}

          - uses: actions/setup-node@v4
            with:
              node-version: ${{ env.NODE_VERSION }}

          - name: Enable KVM
            run: |
              echo 'KERNEL=="kvm", GROUP="kvm", MODE="0666", OPTIONS+="static_node=kvm"' | sudo tee /etc/udev/rules.d/99-kvm4all.rules
              sudo udevadm control --reload-rules
              sudo udevadm trigger --name-match=kvm

          - name: Install Maestro
            uses: dniHze/maestro-test-action@v1

          - name: Install dependencies
            run: bun install --frozen-lockfile || bun install

          - name: Generate native Android project
            run: bunx expo prebuild --platform android --clean --no-install
            env:
              EXPO_PUBLIC_CONVEX_URL: https://placeholder.convex.cloud

          - name: Bump Gradle JVM memory
            run: |
              echo "org.gradle.jvmargs=-Xmx4g -XX:MaxMetaspaceSize=512m" >> android/gradle.properties
              echo "org.gradle.parallel=true" >> android/gradle.properties

          - name: Make gradlew executable
            run: chmod +x android/gradlew

          - uses: gradle/actions/setup-gradle@v4

          - name: Build debug APK
            run: cd android && ./gradlew app:assembleDebug --no-daemon
            env:
              EXPO_PUBLIC_CONVEX_URL: https://placeholder.convex.cloud

          - name: Run Maestro E2E tests
            uses: reactivecircus/android-emulator-runner@v2
            with:
              api-level: 30
              arch: x86_64
              target: default
              emulator-boot-timeout: 600
              script: |
                adb install android/app/build/outputs/apk/debug/app-debug.apk
                maestro test .maestro/ --format junit --output maestro-report.xml

          - name: Upload Maestro report
            if: always()
            uses: actions/upload-artifact@v4
            with:
              name: maestro-report
              path: maestro-report.xml
              retention-days: 14
    ```

    Make sure the .maestro/ directory is created with both `shared/` subdirectory and all flow files.
