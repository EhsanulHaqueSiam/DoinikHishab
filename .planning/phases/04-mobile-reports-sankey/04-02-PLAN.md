import React from "react";
import { Text, View } from "react-native";

import { useTranslation } from "react-i18next";
import { formatCurrency } from "../../lib/currency";

interface NetWorthLineChartProps {

}

export function NetWorthLineChart({ data }: NetWorthLineChartProps) {
  const { t } = useTranslation();
  const currentValue = data.length > 0 ? data[data.length - 1].netWorth : 0;
  const firstValue = data.length > 0 ? data[0].netWorth : 0;
  const trend = currentValue >= firstValue ? "up" : "down";
  const trendDiff = currentValue - firstValue;

  // Convert to gifted-charts data points (value in taka)
  const lineData = data.map((d) => ({
    value: d.netWorth / 100,
    label: formatMonth(d.month),
    dataPointText: "", // no inline labels
  }));

  // Use LineChart with: areaChart, curved, isAnimated
  // color="#14b8a6" (line stroke), thickness={2}
  // startFillColor="#0d9488", startOpacity={0.3}, endFillColor="transparent", endOpacity={0}
  // dataPointsColor="#14b8a6", dataPointsRadius={4}
  // Current value header: text-xl font-bold, conditional color (foreground if positive, danger if negative)
  // Trend indicator: arrow + amount, top-right
  // Card wrapper with accent glow: border border-accent-300/10, shadow("#e6a444", 0, 4, 0.08, 16)
}

function formatMonth(month: string): string {
  const [year, m] = month.split("-");
