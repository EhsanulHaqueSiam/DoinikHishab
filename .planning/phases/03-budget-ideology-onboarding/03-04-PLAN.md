    ```tsx

    ```

    **4. Implement redirect via useEffect** — Use Expo Router's `useRouter` and `useSegments` to redirect:

    ```tsx
    import { useRouter, useSegments } from "expo-router";

    // Inside RootLayout, after loaded check:
    const router = useRouter();
    const segments = useSegments();
    const onboardingDone = isOnboardingComplete();

    useEffect(() => {
      if (loaded && !onboardingDone && !segments[0]?.startsWith("onboarding")) {
        router.replace("/onboarding/rules");
      }
    }, [loaded, onboardingDone, segments]);
    ```

    **5. Keep all existing Stack.Screen registrations intact.** Only ADD the onboarding screen and the redirect logic. Do not remove or modify any existing screens.
